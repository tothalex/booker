19  public class App extends Application
20  {
21  
22      private double xOffset = 0;
23      private double yOffset = 0;
24  
25      @Override
26      public void start(Stage primaryStage) throws Exception{
27          Parent root = FXMLLoader.load(getClass().getResource("/scheme.fxml"));
28          primaryStage.setTitle("booker");
29          primaryStage.setScene(new Scene(root, 750, 400));
30          primaryStage.initStyle(StageStyle.UNDECORATED);
31          primaryStage.show();
32  
33          root.setOnMousePressed(new EventHandler<MouseEvent>() {
34              @Override
35              public void handle(MouseEvent event) {
36                  xOffset = event.getSceneX();
37                  yOffset = event.getSceneY();
38              }
39          });
40          root.setOnMouseDragged(new EventHandler<MouseEvent>() {
41              @Override
42              public void handle(MouseEvent event) {
43                  primaryStage.setX(event.getScreenX() - xOffset);
44                  primaryStage.setY(event.getScreenY() - yOffset);
45              }
46          });
47      }
48  
49      public static void main(String[] args) throws MessagingException {
50          launch(args);
51      }
